    private boolean TryConnect()
    {
        try 
        {
            Class.forName("com.mysql.jdbc.Driver");
            con = (Connection) DriverManager.getConnection(url, user_name, password);
            return true;
        } 
        catch (SQLException ex) 
        {
            System.out.println(ex);
            return false;
        }
        catch (ClassNotFoundException ex) 
        {
            System.out.println(ex);
            return false;
        }
    }
